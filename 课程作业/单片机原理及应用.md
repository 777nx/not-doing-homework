## 示例代码

实验二文档参考代码

```c
/*--------------------------------------------------------------------
第一次按下按键K0后，发光二极管D0点亮；再次按下按键K0后，D0熄灭，如此循环
----------------------------------------------------------------------*/
#include <reg51.h>
#define uchar unsigned char // 宏定义
#define uint unsigned int
	
sbit D0 = P2 ^ 0;
sbit K0 = P1 ^ 0;

void delayms(uint y) {
	uint i, j;
	for(i = y; i > 0; i--)
		for(j = 110; j > 0; j--)
			;
}

main(void) {
	D0 = 1; // D0初始是灭的
	while(1) {
		if(K0 == 0) { // 判断按键是否被按下
			delayms(10); // 延时10ms
			if(K0 == 0) { // 再次判断按键是否被按下
				D0 =! D0; // 确认按键是被按下，D0状态取反
			}
			while(!K0); // 等待按键释放（松手）
		}
	}
}
```
